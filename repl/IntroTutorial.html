Customizing the Model¶

Up to this point we used native PyTorch tools. However, we can improve upon the performance by utilizing specialized modules. These modules have useful functions such as localized weight regularization, null-adjusted log likelihood, and automatic weight reshaping.

In addition, they include implementations for useful optimizers such as LBFGS which is a second order method that converges much more quickly than iterative methods (gradient descent) for simple models.

Please Read the following alternative implementation of the model you built in Task 1. Feel free to explore the implementations for the imported tools from the local NDNT module.

class GLM(nn.Module):

    def __init__(self,
                 intput_dims,
                 NC,
                 reg_vals = None):

        super().__init__()

        self.dims = intput_dims
        self.cids = np.arange(NC)
        self.loss = PoissonLoss_datafilter()

        # build model
        self.layer = NDNLayer(input_dims=self.dims,
                              num_filters=NC, reg_vals=reg_vals, bias=True,
                              NLtype='softplus')

    def forward(self, input):

        y = self.layer(input['stim'])

        return y

    def prepare_regularization(self, normalize_reg=False):

        self.layer.reg.normalize = normalize_reg
        self.layer.reg.build_reg_modules()

    def compute_reg_loss(self):

        return self.layer.compute_reg_loss()

    def training_step(self, batch, batch_idx=None):  # batch_indx not used, right?

        y = batch['robs']
        dfs = batch['dfs']

        y_hat = self(batch)

        loss = self.loss(y_hat, y, dfs)

        regularizers = self.compute_reg_loss()

        return {'loss': loss + regularizers, 'train_loss': loss, 'reg_loss': regularizers}

    def validation_step(self, batch, batch_idx=None):

        y = batch['robs']
        dfs = batch['dfs']

        y_hat = self(batch)

        loss = self.loss(y_hat, y, dfs)

        return {'loss': loss, 'val_loss': loss, 'reg_loss': None}

Evaluate Model Using Null-Adjusted LL

We can use the following function for evaluating our model.

In [ ]:
def eval_model(model, valid_dl):
    loss = model.loss.unit_loss
    model.eval()

    LLsum, Tsum, Rsum = 0, 0, 0
    from tqdm import tqdm

    device = next(model.parameters()).device  # device the model is on
    if isinstance(valid_dl, dict):
        for dsub in valid_dl.keys():
                if valid_dl[dsub].device != device:
                    valid_dl[dsub] = valid_dl[dsub].to(device)
        rpred = model(valid_dl)
        LLsum = loss(rpred,
                    valid_dl['robs'][:,model.cids],
                    data_filters=valid_dl['dfs'][:,model.cids],
                    temporal_normalize=False)
        Tsum = valid_dl['dfs'][:,model.cids].sum(dim=0)
        Rsum = (valid_dl['dfs'][:,model.cids]*valid_dl['robs'][:,model.cids]).sum(dim=0)

    else:
        for data in tqdm(valid_dl, desc='Eval models'):

            for dsub in data.keys():
                if data[dsub].device != device:
                    data[dsub] = data[dsub].to(device)

            with torch.no_grad():
                rpred = model(data)
                LLsum += loss(rpred,
                        data['robs'][:,model.cids],
                        data_filters=data['dfs'][:,model.cids],
                        temporal_normalize=False)
                Tsum += data['dfs'][:,model.cids].sum(dim=0)
                Rsum += (data['dfs'][:,model.cids] * data['robs'][:,model.cids]).sum(dim=0)

    LLneuron = LLsum/Rsum.clamp(1)

    rbar = Rsum/Tsum.clamp(1)
    LLnulls = torch.log(rbar)-1
    LLneuron = -LLneuron - LLnulls

    LLneuron/=np.log(2)

    return LLneuron.detach().cpu().numpy()

class LNLN(nn.Module):

    def __init__(self, input_dims, num_subunits, NC, reg_vals = None):

        super().__init__()
        self.dims = input_dims
        self.num_subunits = num_subunits
        self.cids = np.arange(NC)
        self.loss = PoissonLoss_datafilter()

        # build model
        self.layer0 = NDNLayer(input_dims=self.dims,
                              num_filters=num_subunits, reg_vals=reg_vals, bias=True,
                              NLtype='relu')
        self.layer1 = NDNLayer(input_dims=[num_subunits, 1, 1, 1],
                              num_filters=NC, bias=True,
                              NLtype='softplus')

    def forward(self, input):

        y = self.layer0(input['stim'])
        return self.layer1(y)

    def prepare_regularization(self, normalize_reg=False):

        self.layer0.reg.normalize = normalize_reg
        self.layer1.reg.normalize = normalize_reg

        self.layer0.reg.build_reg_modules()
        self.layer1.reg.build_reg_modules()

    def compute_reg_loss(self):

        return self.layer0.compute_reg_loss() + self.layer1.compute_reg_loss()

    def training_step(self, batch, batch_idx=None):  # batch_indx not used, right?

        y = batch['robs']
        dfs = batch['dfs']

        y_hat = self(batch)

        loss = self.loss(y_hat, y, dfs)

        regularizers = self.compute_reg_loss()

        return {'loss': loss + regularizers, 'train_loss': loss, 'reg_loss': regularizers}

    def validation_step(self, batch, batch_idx=None):

        y = batch['robs']
        dfs = batch['dfs']

        y_hat = self(batch)

        loss = self.loss(y_hat, y, dfs)

        return {'loss': loss, 'val_loss': loss, 'reg_loss': None}

def train_eval(glm0):
    glm0.prepare_regularization()
    glm0.to(device)

    optimizer = torch.optim.LBFGS(glm0.parameters(),
                    history_size=10,
                    max_iter=10e4,
                    tolerance_change=1e-9,
                    line_search_fn=None,
                    tolerance_grad=1e-5)

    trainer = LBFGSTrainer(
        optimizer=optimizer,
        device=device,
        accumulate_grad_batches=len(train_dataloader),
        max_epochs=1,
        optimize_graph=True,
        log_activations=False,
        set_grad_to_none=False,
        verbose=0)

    trainer.fit(glm0, train_dataloader.dataset[:], val_dataloader.dataset[:])
    s = eval_model(glm0, test_dataloader).mean()
    del optimizer, trainer
    return s

def eval_LNLN(num_hidden):

    s = train_eval(glm0)
    del glm0
    return s

def eval_GLM():
    glm0 = GLM(dims, NC, reg_vals={'glocalx': 10, 'd2t':10})
    s = train_eval(glm0)
    del glm0
    return s
